@breakpoint(handler=printer, dt=1.0)
def counter(n):
    result = 0
    alarm = Alarm()
    for _ in range(n):
        if alarm.next():
            progress = result / n
            alarm.update((yield progress, result))
        time.sleep(0.1); result = result + 1
    yield 1.0, result
